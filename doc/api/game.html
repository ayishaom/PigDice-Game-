 16def dice_face(n: int) -> str:
 17    """Return a Unicode character representing a dice face for numbers 1–6.
 18
 19    Args:
 20        n (int): Number rolled on the dice.
 21
 22    Returns:
 23        str: Unicode ⚀–⚅ for 1–6, or str(n) if out of range.
 24    """
 25    if 1 <= n <= 6:
 26        return chr(DICE_FACE_BASE + n - 1)
 27    return str(n)

 93        """Prompt the human player for a command and return it.

 98        valid = {

114            except (EOFError, KeyboardInterrupt):

158    def play(self):
159        """Run the main game loop until a player wins or quits.
160
161        Handles turn switching, rolling, holding, cheating, AI decisions, and score updates.
162        """
163        self.running = True
164        while self.running:
165            self.show_board()
166            current = self.players[self.current_index]
167            opponent = self.players[1 - self.current_index]
168            self.turn_total = 0
169
170            turn_header = "🤖  " + current.name if current.is_ai else "👤  " + current.name
171            print(f"\n--- {turn_header}'s turn ---  (type 'help' for commands)\n")
172
173            turn_active = True
174
175            while turn_active and self.running:
176                if current.is_ai:
177                    decision = self.ai_agent.decide(self.turn_total, current.total_score, opponent.total_score)
178                    print(f"[🤖 AI decides to {decision}]")
179                else:
180                    cmd = self._prompt_cmd()
181                    decision = cmd
182                    if decision == "help":
183                        print(self._help_text())
184                        continue
185                    if decision == "quit":
186                        print("\n👋 Quitting current game and returning to menu...\n")
187                        self.running = False
188                        return
189                    if decision == "restart":
190                        print("\n🔄 Restarting current match (scores reset).\n")
191                        for p in self.players:
192                            p.reset_score()
193                        turn_active = False
194                        break
195                    if decision == "ai":
196                        if any(p.is_ai for p in self.players):
197                            level = self._prompt_ai_level()
198                            if level:
199                                try:
200                                    self.ai_agent.set_difficulty(level)
201                                    print(f"🔧 AI difficulty set to {level}.")
202                                except Exception as e:
203                                    print("⚠️  Invalid difficulty:", e)
204                        else:
205                            print("ℹ️  AI difficulty can only be changed when an AI opponent exists.")
206                        continue
207                    if decision == "name":
208                        try:
209                            new_name = input("➤ Enter new name [Enter to cancel]: ")
210                        except (EOFError, KeyboardInterrupt):
211                            print("\n❌ Name change cancelled.")
212                            continue
213                        new_name = (new_name or "").strip()
214                        if not new_name:
215                            print("❌ Name change cancelled.")
216                            continue
217                        old = current.name
218                        renamed = self.score_manager.rename_player(old, new_name)
219                        current.set_name(new_name)
220                        if renamed:
221                            print(f"✍️  Renamed {old} → {new_name} and preserved stats.")
222                        else:
223                            print(f"✍️  Renamed locally to {new_name}. No previous stats existed for {old}.")
224                        continue
225
226                # Handle roll/hold/cheat
227                if decision == "roll":
228                    values = self.dice_hand.roll()
229                    roll = values[0] if values else None
230                    face = dice_face(roll) if roll is not None else "?"
231                    print(f"🎲 {current.name} rolled {roll} {face}")
232
233                    if roll is None:
234                        print("Invalid roll. Turn lost.")
235                        self.turn_total = 0
236                        turn_active = False
237                        continue
238
239                    if roll == 1:
240                        print("💥  Oh no — rolled a 1. Turn lost.")
241                        self.turn_total = 0
242                        turn_active = False
243                    else:
244                        self.turn_total += roll
245                        print(f"➕ Turn total is now {self.turn_total}. (Hold to bank points)")
246                        if current.total_score + self.turn_total >= self.winning_score:
247                            current.add_score(self.turn_total)
248                            print(f"\n🏆 {current.name} reaches {current.total_score} points and wins!\n")
249                            self.score_manager.record_game(current.name, current.total_score)
250                            return
251
252                elif decision == "hold":
253                    current.add_score(self.turn_total)
254                    print(f"💰 {current.name} banks {self.turn_total} points (total {current.total_score}).")
255                    if current.total_score >= self.winning_score:
256                        print(f"\n🏆 {current.name} wins with {current.total_score} points!\n")
257                        self.score_manager.record_game(current.name, current.total_score)
258                        return
259                    turn_active = False
260
261                elif decision == "cheat":
262                    cheat_points = 50
263                    print(f"🪄 Cheat used! Adding {cheat_points} points to {current.name}.")
264                    current.add_score(cheat_points)
265                    if current.total_score >= self.winning_score:
266                        print(f"\n🏆 {current.name} wins by cheat with {current.total_score} points!\n")
267                        self.score_manager.record_game(current.name, current.total_score)
268                        return
269                    turn_active = False
270
271            self.current_index = 1 - self.current_index

168            opponent = self.players[1 - self.current_index]
169            self.turn_total = 0
170
171            turn_header = "🤖  " + current.name if current.is_ai else "👤  " + current.name
172            print(f"\n--- {turn_header}'s turn ---  (type 'help' for commands)\n")
173
174            turn_active = True
175
176            while turn_active and self.running:
177                if current.is_ai:
178                    decision = self.ai_agent.decide(self.turn_total, current.total_score, opponent.total_score)
179                    print(f"[🤖 AI decides to {decision}]")
180                else:
181                    cmd = self._prompt_cmd()
182                    decision = cmd
183                    if decision == "help":
184                        print(self._help_text())
185                        continue
186                    if decision == "quit":
187                        print("\n👋 Quitting current game and returning to menu...\n")
188                        self.running = False
189                        return
190                    if decision == "restart":
191                        print("\n🔄 Restarting current match (scores reset).\n")
192                        for p in self.players:
193                            p.reset_score()
194                        turn_active = False
195                        break
196                    if decision == "ai":
197                        if any(p.is_ai for p in self.players):
198                            level = self._prompt_ai_level()
199                            if level:
200                                try:
201                                    self.ai_agent.set_difficulty(level)
202                                    print(f"🔧 AI difficulty set to {level}.")
203                                except Exception as e:
204                                    print("⚠️  Invalid difficulty:", e)
205                        else:
206                            print("ℹ️  AI difficulty can only be changed when an AI opponent exists.")
207                        continue
208                    if decision == "name":
209                        try:
210                            new_name = input("➤ Enter new name [Enter to cancel]: ")
211                        except (EOFError, KeyboardInterrupt):
212                            print("\n❌ Name change cancelled.")
213                            continue
214                        new_name = (new_name or "").strip()
215                        if not new_name:
216                            print("❌ Name change cancelled.")
217                            continue
218                        old = current.name
219                        renamed = self.score_manager.rename_player(old, new_name)
220                        current.set_name(new_name)
221                        if renamed:
222                            print(f"✍️  Renamed {old} → {new_name} and preserved stats.")
223                        else:
224                            print(f"✍️  Renamed locally to {new_name}. No previous stats existed for {old}.")
225                        continue
226
227                # Handle roll/hold/cheat
228                if decision == "roll":
229                    values = self.dice_hand.roll()
230                    roll = values[0] if values else None
231                    face = dice_face(roll) if roll is not None else "?"
232                    print(f"🎲 {current.name} rolled {roll} {face}")
233
234                    if roll is None:
235                        print("Invalid roll. Turn lost.")
236                        self.turn_total = 0
237                        turn_active = False
238                        continue
239
240                    if roll == 1:
241                        print("💥  Oh no — rolled a 1. Turn lost.")
242                        self.turn_total = 0
243                        turn_active = False
244                    else:
245                        self.turn_total += roll
246                        print(f"➕ Turn total is now {self.turn_total}. (Hold to bank points)")
247                        if current.total_score + self.turn_total >= self.winning_score:
248                            current.add_score(self.turn_total)
249                            print(f"\n🏆 {current.name} reaches {current.total_score} points and wins!\n")
250                            self.score_manager.record_game(current.name, current.total_score)
251                            return
252
253                elif decision == "hold":
254                    current.add_score(self.turn_total)
255                    print(f"💰 {current.name} banks {self.turn_total} points (total {current.total_score}).")
256                    if current.total_score >= self.winning_score:
257                        print(f"\n🏆 {current.name} wins with {current.total_score} points!\n")
258                        self.score_manager.record_game(current.name, current.total_score)
259                        return
260                    turn_active = False
261
262                elif decision == "cheat":
263                    cheat_points = 50
264                    print(f"🪄 Cheat used! Adding {cheat_points} points to {current.name}.")
265                    current.add_score(cheat_points)
266                    if current.total_score >= self.winning_score:
267                        print(f"\n🏆 {current.name} wins by cheat with {current.total_score} points!\n")
268                        self.score_manager.record_game(current.name, current.total_score)
269                        return
270                    turn_active = False
271
272            self.current_index = 1 - self.current_index

Run the main game loop until a player wins or quits.
